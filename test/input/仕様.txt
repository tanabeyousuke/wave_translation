module_num: !特定ユニットの数
  	    lfo 1; !lfoの個数
  	    efc 3;

setting:
  osc sin 30; !オシレータの音量
  osc del 30;

  env atk r5;
  env dec r6;
  env sus r7;
  env rel r8;
  env out 399;

  lfo sqr r1 r9 r10; !波形、周波数、振幅、オフセット、出力 

  efc low 3000 15; !エフェクト 入力数は種類によって異なる
  efc hig 5000 20;
  efc bit 15;
  
  amp r5; !最終的にどれぐらいの大きさで音を出すか決めるアンプ

play:
  mov r10 1.0; !レジスタへ代入 この場合、r10はlfoのsinの振幅につながっているので、振幅が1.0になる
  ;... !他のも色々代入しなきゃいけないけど、中略
	
  spd 192; !速度記号 この場合、四分音符が一分間に192回入る速さ
  dis 4 4p; !音符 この場合、第四オクターブdisの付点四分音符
  rst 1; !休符 この場合、全休符

  mov r6 3.1; !レジスタの値はいつでも変えられるが、この例で言うところのr1など、lfoの出力などは代入後にすぐ変えられる可能性がある

  ... !音符とかをもっと書いてく
  !音割れを防ぐために実際の出力の値は ampの値/シンセサイザの数にする予定
